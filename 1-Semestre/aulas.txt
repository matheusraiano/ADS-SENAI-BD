Aula(12/03/2026):
    Utilização do Xampp e MySQL Workbench para manipular o banco de dados
    Comandos SQL:
        SHOW DATABASES; --> lista todos os bancos de dados que existem no sistema
        CREATE DATABASE 'banco'; --> cria um banco de dados
        DROP DATABASE 'banco'; --> tirar alguma coisa do banco, remove da existencia
        CREATE SCHEMA 'banco'; --> cria uma database do mesmo jeito
        DROP SCHEMA 'banco'; --> dropa do mesmo jeito
        USE 'banco' --> deixa o banco em uso para mexer nele
        CREATE TABLE 'tabela' --> cria uma tabela no banco usado
    Tipos de comandos:
        DDL: Data Definition Language
            CREATE
            DROP
            ALTER
            TRUNCATE
            COMMENT
        DQL: Data Query Language
            SELECT
        DML: Data Manipulation Language
            INSERT
            UPDATE
            DELETE
        DFL: Data Transaction Language
            BEGIN TRANSACTION
            COMMIT
            ROLLBACK
    Tipos de dados:
        Tipos Numéricos:
            bit ou bool/boolean --> ocupa 1 byte de memoria
            tinyint --> ocupa 1 byte de memoria, aceita valores de -128 até 127
            smallint --> ocupa 2 bytes de memoria, aceita valores de -32.768 até 32.767
            mediumint --> ocupa 3 bytes de memoria, aceita valores de -8.388.608 até 8.388.607
            integer/int --> ocupa 4 bytes de memoria, aceita valores de -2.147.483.648 até 2.147.483.647
            bigint --> ocupa 8 bytes de memoria, aceita valores de -9.223.372.036.854.775.808 até 9.223.372.036.854.775.807
            float --> ocupa 4 bytes de memoria, aceita no máximo até 6 casas decimais depois da vírgula(123.456), caso não seja especificado dentro do (), ou seja posso passar apenas um parametro, exemplo (3) fica 123
            double --> ocupa 8 bytes de memoria, aceita no máximo até 24 casas decimais depois da vírgula, caso não especificado dentro do (), ou seja posso passar dois parametros, exemplo: (6,2) fica 1234.56 -> 6 casas com dois pontos flutuantes
            decimal --> ocupa o primeiro valor do parametro mais 2 bytes de memoria, sem parametro utiliza precisao máxima, ou seja posso usar dois parametros, exemplo: (7,2) fica 12345.67 -> 7 cassas com dois pontos flutuantes
        Cadeia de Caracteres:
            char --> ocupa quantidade de caracteres vezes 1 byte de memória, sózinho(sem especificar) ele ocupa o máximo, utilizar quando tem certeza da quantidade de caracteres, tem um parametro sendo ele o tamanho, exemplo: (11) fica 11 caracteres
            varchar(n) --> ocupa n + 1 byte de memória, o que colocar no n mostra o máximo que ele pode ocupar, exemplo: (65) fica 65 caracteres e 66 bytes
            tinytext --> ocupa o espaço máximo de memória caso sozinho, é sempre o tamanho + 1 byte e é limitado até 256 caracteres
            text --> ocupa o espaço máximo de memória caso sozinho, é sempre o tamanho + 1 byte e é limitado até até 65.535 caracteres, aceita caracteres do tipo UTF
            mediumtext --> ocupa o espaço máximo de memória caso sozinho, é sempre o tamanho + 1 byte e é limitado até até 16.777.215 caracteres, aceita caracteres do tipo UTF
            enum --> suporta lista de até 64 itens, utilizado para armazenar lista de valores
            set --> suporta lista de até 65 mil itens, utilizado para armazenar lista de valores
        Datas:
            date --> armazena uma data considerando 01/01/0001 até 31/12/9999, ocupa 3 bytes de memória
            datetime --> dia, mes, ano e hora com minutos e segundos considerando 01/01/0001 até 31/12/9999, ocupa 8 bytes de memória 
            timestamp --> dia, mes, ano e hora com minutos e segundos considerando 01/01/1970 até 19/01/2038, ocupa 4 bytes de memória 
            time --> ocupa 3 bytes de memória, aceita horas minutos e segundos
            year --> ocupa 1 byte de memória, pode passar parametros, sendo o padrao de 4 digitos, podemos passar de 2 ou 4 no parametros para digitos, armazena de 1901 até 2155
        Tipos Blob:
            tinyblob --> ocupa 8 bits
            blob --> ocupa 16 bits
            mediumblob --> ocupa 24 bits
            longblob --> ocupa 32 bits

    constraints:
        NOT NULL --> não deixa passar sem a informação
        NULL --> deixa passar sem a informação, é o padrão
        DEFAULT --> cria dentro da coluna a informação colocada
        UNIQUE --> é um campo unico
        CHECK --> checagem dentro do () que for definido
        PRIMARY KEY --> chave primária
        AUTO_INCREMENT --> não precisa mais rodar o campo em si, pois ele vai se incrementando a cada vez que o código rodar
            é um contador interno, logo se deletar o luiz 3 e criar a kamila ela vai ser 4 e não 3, ficando com uma lacuna 
        SERIAL DEFAULT VALUE--> PRIMARY KEY + DEFAULT + NOT NULL
        FOREIGN KEY(coluna da tabela estrangeira) --> chave estrangeira
            REFERENCES(tabela) --> refencia que fa com a tabela selecionada
    para criar uma constraint com nome: CONSTRAINT 'nome' 'constraint para usar' ('tabela')
    quando tiver tabela que depende de outra, sempre criar primeiro a que não depende de alguma e vai nesse sentido

    comandos:
        UNION --> justa selects, fazendo isso pode-se tirar os nulos quando executa um select com varias colunas e retira os repetidos. Caso eu queira que não retire só usar union all
            SELECT 'cliente', clienteId from tb_endereco
            UNION
            SELECT 'funcionario', funcionarioId from tb_endereco
            UNION
            SELECT 'fornecedor', fornecedorId from tb_endereco;
        INNER JOIN --> entra em outra tabela dentro de um select de outra tabela. Tabela que vai no from é a tabela da esquerda, a que vai depois é a da direita!!! o inner mostra o que faz referência com o que tem na da esquerda e da direita
            SELECT c.descricao 'categoria', p.descricao 'produto', p.preco - (p.preco * 0.02) 'c/ desconto' 
            FROM tb_produto p
            INNER JOIN tb_categoria c
            ON p.categoriaId = c.categoriaId
            WHERE c.descricao = 'Bebidas'
        LEFT JOIN --> ele mostra tudo da tabela da esqurda + o que tem de relacional entre as tabelas e da direta apenas o que está relacionado
        RIGHT JOIN --> ele mostra tudo da tabela da direita + o que tem de relacional entre as tabelas e da esquerda apenas o que está relacionado
        FULL JOIN --> mostra tudo que está relacionao ou não
